class Data:
    def __init__(self, file_path):
        self.file_path = file_path
        
    def load_csv(self):
        # Handle errors if something goes wrong
        try:
            self.data = pd.read_csv(self.file_path)
            print("CSV file laoded successfully!")
        
        except FileNotFoundError:
            print("Error: The file was not found")
        
        except Exception as e:
            print(f"Error while loading file: {e}")
        
        return self.data

class DataProcessor:
    
    # Loads the CSV file so you can work with the dataset
    def __init__(self, file_path):
        self.data = pd.read_csv(file_path)
    
    # Counts how many columns (features) are in the dataset
    def column_count(self):
        return self.data.shape[1]
    
   # Get first 5 rows if no argument is pass
    def head(self, n=5):
        return self.data.head(n)
    
    # Get last 5 rows if no argument is pass
    def tail(self, n=5):
        return self.data.tail(n)
    
    # Tells how many rows and columns are in the dataset
    def shape(self):
        return self.data.shape
    
    # Randomly shows a few rows to spot-check the dat
    def sample(self, n=5):
        return self.data.sample(n)
    
    # Gives an overview of each column’s data type and how many missing values
    def info(self):
        return self.data.info()
    
    # Gives basic statistics like mean, std, min, max for numeric columns
    def describe(self):
        return self.data.describe()
    
    #  Helps retrieve specific row
    def loc_rows(self, start, end):
        return self.data.loc[start:end]
    
    #  Helps retrieve specific row
    def iloc_rows(self, start, end):
        return self.data.iloc[start:end]
    
    # Shows central values for numeric data
    def get_mean(self):
        return self.data.mean(numeric_only=True)
    
    # Shows central values for numeric data
    def get_median(self):
        return self.data.median(numeric_only=True)
    
    # Shows central values for numeric data
    def get_mode(self):
        return self.data.mode()
    # Checks for missing values in the dataset
    def is_null(self):
        return self.data.isnull().sum()
    # Groups the data by one column and shows the average values per group
    def group_by(self, column_name):
        return self.data.groupby(column_name).mean(numeric_only=True)

class Graph:
    def __init__(self, data):
        self.data = data
    
    # Helps compare average values across categories
    def bar_plot(self, x, y, hue=None):
        plt.figure(figsize=(10, 6))
        sns.barplot(x=x, y=y, hue=hue, data=self.data)
        plt.title(f'Bar Plot: {y} by {x}')
        plt.xticks(rotation=45)
        plt.tight_layout()
        plt.show()
    
    # Great for seeing frequency of categories
    def count_plot(self, column):
        plt.figure(figsize=(8, 5))
        sns.countplot(x=column, data=self.data)
        plt.title(f'Count Plot of {column}')
        plt.xticks(rotation=45)
        plt.tight_layout()
        plt.show()
    
    # Shows distribution of numeric data like price or area
    def hist_plot(self, column):
        plt.figure(figsize=(8, 5))
        sns.histplot(self.data[column], kde=True)
        plt.title(f'Histogram of {column}')
        plt.tight_layout()
        plt.show()
    
    # Detects outliers, medians, and variation in different groups
    def box_plot(self, x, y):
        plt.figure(figsize=(10, 6))
        sns.boxplot(x=x, y=y, data=self.data)
        plt.title(f'Box Plot of {y} by {x}')
        plt.xticks(rotation=45)
        plt.tight_layout()
        plt.show()
    
    # Detects outliers, medians, and variation in different groups
    def scatter_plot(self, x, y):
        plt.figure(figsize=(8, 5))
        sns.scatterplot(x=x, y=y, data=self.data)
        plt.title(f'Scatter Plot: {y} vs {x}')
        plt.tight_layout()
        plt.show()
    
    # Helps detect strong relationships between features for modeling
    def heatmap(self):
        plt.figure(figsize=(12, 8))
        sns.heatmap(self.data.corr(numeric_only=True), annot=True, cmap='coolwarm')
        plt.title('Correlation Heatmap')
        plt.tight_layout()
        plt.show()

class BivariateAnalysis:
    def __init__(self, data):
        self.data = data
        self.graph = Graph(data)  # Graph class ka object bna liya

    def price_vs_area_type(self):
        print("Box Plot: Price vs Area Type")
        self.graph.box_plot("area_type", "price")

    def price_vs_location(self):
        print("Bar Plot: Avg Price by Location")
        top_locations = self.data['location'].value_counts().head(10).index
        filtered_data = self.data[self.data['location'].isin(top_locations)]
        graph_local = Graph(filtered_data)
        graph_local.bar_plot("location", "price")

    def sqft_vs_price(self):
        print("Scatter Plot: Price vs Total Sqft")
        self.graph.scatter_plot("total_sqft", "price")

    def bhk_vs_price(self):
        print("Box Plot: Price vs BHK")
        if 'bhk' in self.data.columns:
            self.graph.box_plot("bhk", "price")
        else:
            print("Column 'bhk' not found!")

    def show_correlation(self):
        print("Heatmap: Numeric Correlation")
        self.graph.heatmap()

class Model:
    def __init__(self, data, target_column):
        self.data = data
        self.target_column = target_column
        self.model = SVC()
        self.scaler = StandardScaler()

    def split_data(self, test_size=0.2, random_state=42):
        X = self.data.drop(self.target_column, axis=1)
        y = self.data[self.target_column]
        self.X_train, self.X_test, self.y_train, self.y_test = train_test_split(
            X, y, test_size=test_size, random_state=random_state)
        print("Data split done.")

    def scale_data(self):
        self.X_train = self.scaler.fit_transform(self.X_train)
        self.X_test = self.scaler.transform(self.X_test)
        print("Feature scaling complete.")

    def train_model(self):
        self.model.fit(self.X_train, self.y_train)
        print("Model training complete.")

    def predict(self):
        self.y_pred = self.model.predict(self.X_test)
        return self.y_pred

    def evaluate(self):
        print("Accuracy:", accuracy_score(self.y_test, self.y_pred))
        print("\nConfusion Matrix:\n", confusion_matrix(self.y_test, self.y_pred))
        print("\nClassification Report:\n", classification_report(self.y_test, self.y_pred))

Pickle Class:¶

init(self, model=None): Prepares the class to save or load a model later¶

save_model(filename='svm_model.pkl'): Allows to store the trained model on disk for future use without retraining¶

load_model(filename='svm_model.pkl'): Lets reuse a previously trained model anytime by loading it back¶

predict_new(input_data): Enables making predictions on new, unseen data using the saved model¶

In [79]:
class PickleData:
    def __init__(self, model=None):
        self.model = model

    def save_model(self, filename='svm_model.pkl'):
        with open(filename, 'wb') as file:
            pickle.dump(self.model, file)
        print(f"Model saved as {filename}")

    def load_model(self, filename='svm_model.pkl'):
        with open(filename, 'rb') as file:
            self.model = pickle.load(file)
        print(f"Model loaded from {filename}")

    def predict_new(self, input_data):
        if self.model is None:
            print("Model is not loaded!")
            return None
        input_array = np.array(input_data).reshape(1, -1)
        prediction = self.model.predict(input_array)
        print("Prediction:", prediction)
        return prediction
